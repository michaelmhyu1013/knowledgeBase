CREATE TABLE users (
    "password" varchar(128) NOT NULL,
    imageurl varchar(128),
    country varchar(50),
    description varchar(250),
    dob date,
    messages integer DEFAULT 0, 
    posts integer DEFAULT 0,
    likes integer DEFAULT 0,
    email varchar(128) NOT NULL,
    userid serial NOT NULL,
    firstname varchar(128) NOT NULL,
    lastname varchar(128) NOT NULL,
    CONSTRAINT user_pk PRIMARY KEY (userid)
);


CREATE TABLE sessions (
   userid integer NOT NULL,
   sessionid VARCHAR(256) NOT NULL,
   CONSTRAINT session_pk PRIMARY KEY (sessionid),
   CONSTRAINT userid_fk FOREIGN KEY (userid) REFERENCES users (userid)
);

CREATE TABLE threads (
    threadid serial NOT NULL,
    userid integer NOT NULL,
    title varchar(50) NOT NULL,
    body varchar(1000) NOT NULL,
    date date NOT NULL,
    subject varchar(50) NOT NULL,
    replyCount integer DEFAULT 0,
    CONSTRAINT thread_pk PRIMARY KEY (threadid),
	CONSTRAINT userid_fk FOREIGN KEY (userid) REFERENCES users(userid)
);
    
CREATE TABLE posts (
    postid serial NOT NULL,
    threadid integer NOT NULL,
    userid integer NOT NULL,
    title varchar(50) NOT NULL,
    body varchar(1000) NOT NULL,
    date date NOT NULL,
    CONSTRAINT post_pk PRIMARY KEY (postid, threadid),
	CONSTRAINT threadid_fk FOREIGN KEY (threadid) REFERENCES threads(threadid),
    CONSTRAINT userid_fk FOREIGN KEY (userid) REFERENCES users(userid)
);

CREATE TABLE conversations (
    conversationid serial NOT NULL,
    subject1id integer NOT NULL,
    subject2id integer NOT NULL,
    conversationsubject varchar(250),
    CONSTRAINT convo_pk PRIMARY KEY (conversationid, subject1id, subject2id)
);


CREATE TABLE messages (
    messageid serial NOT NULL,
	subject1id integer NOT NULL,
    subject2id integer NOT NULL,
    conversationid integer NOT NULL,
    body varchar(1000) NOT NULL,
	date date NOT NULL,
    CONSTRAINT message_pk PRIMARY KEY (messageid, conversationid),
	CONSTRAINT convoid_fk FOREIGN KEY (conversationid,subject1id,subject2id) REFERENCES conversations(conversationid,subject1id,subject2id)
);
